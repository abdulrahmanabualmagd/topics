01. color               : [color name | hex value | rgb value | rgba value | hsl value | hsla value]                                                                                :Sets the color of text.
02. background-color    : [color name | hex value | rgb value | rgba value | hsl value | hsla value | transparent]                                                                  :Sets the background color of an element.
03. font-size           : [absolute size | relative size | length | percentage]                                                                                                     :Sets the size of the font.
04. font-family         : [font family name | generic family]                                                                                                                       :Sets the font family for text.
05. font-weight         : [normal | bold | bolder | lighter | 100-900]                                                                                                              :Sets the thickness of the font.
06. margin              : [length | percentage | auto]                                                                                                                              :Sets the margin space around an element.
07. padding             : [length | percentage | auto]                                                                                                                              :Sets the padding space inside an element.
08. border              : [border width | border style | color]                                                                                                                     :Sets the border around an element.
09. width               : [length | percentage | auto | fit-content | min-content | max-content]                                                                                    :Sets the width of an element.
10. height              : [length | percentage | auto | fit-content | min-content | max-content]                                                                                    :Sets the height of an element.
11. display             : [inline | block | inline-block | flex | inline-flex | grid | inline-grid | none]                                                                          :Sets how an element should be displayed.
12. position            : [static | relative | absolute | fixed | sticky]                                                                                                           :Sets the positioning method of an element.
13. float               : [left | right | none]                                                                                                                                     :Specifies whether an element should float left, right, or none.
14. text-align          : [left | right | center | justify]                                                                                                                         :Sets the horizontal alignment of text.
15. text-decoration     : [none | underline | overline | line-through | underline overline | underline line-through | overline line-through | underline overline line-through]      :Sets decorations such as underline, overline, or line-through for text.
